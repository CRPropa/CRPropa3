20// Helper to set POD type methods to variant
21#define VARIANT_ADD_TYPE_DECL_POD(NAME, TYPE, VALUE) \
22 bool is ## NAME() const { return (type == TYPE); } \
23 operator VALUE () const { return to ## NAME(); } \
24 VALUE &as ## NAME() { check(TYPE); return data._##NAME; } \
25 const VALUE &as ## NAME() const { check(TYPE); return data._##NAME; } \
26 static Variant from ## NAME(const VALUE &a) { return Variant(a); } \
27 VALUE to ## NAME() const; \
28 Variant &operator = (const VALUE &a) { clear(); type = TYPE; data._##NAME = a; return *this; } \
29 bool operator != (const VALUE &a) const { check(TYPE); return data._##NAME != a; } \
30 bool operator == (const VALUE &a) const { check(TYPE); return data._##NAME == a; } \
31 Variant(const VALUE &a) { data._ ## NAME = a; type = TYPE; }

40// Helper to set pointer type methods to variant
41#define VARIANT_ADD_TYPE_DECL_PTR(NAME, TYPE, VALUE) \
42 bool operator != (const VALUE &a) const { check(TYPE); return *data._##NAME != a; } \
43 bool operator == (const VALUE &a) const { check(TYPE); return *data._##NAME == a; } \
44 VARIANT_ADD_TYPE_DECL_PTR_BASE(NAME, TYPE, VALUE) \
