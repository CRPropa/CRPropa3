319 T closestValue(const Vector3d &position) const {
320 Vector3d r = (position - gridOrigin) / spacing;
321 int ix, iy, iz;
322 if (reflective) {
323 ix = round(r.x);
324 iy = round(r.y);
325 iz = round(r.z);
326 while ((ix < -0.5) or (ix > (Nx-0.5)))
327 ix = 2 * Nx * (ix > (Nx-0.5)) - ix-1;
328 while ((iy < -0.5) or (iy > (Ny-0.5)))
329 iy = 2 * Ny * (iy > (Ny-0.5)) - iy-1;
330 while ((iz < -0.5) or (iz > (Nz-0.5)))
331 iz = 2 * Nz * (iz > (Nz-0.5)) - iz-1;
332 } else {
333 ix = round(fmod(r.x, Nx));
334 iy = round(fmod(r.y, Ny));
335 iz = round(fmod(r.z, Nz));
336 ix = (ix + Nx * (ix < 0)) % Nx;
337 iy = (iy + Ny * (iy < 0)) % Ny;
338 iz = (iz + Nz * (iz < 0)) % Nz;
339 }
340 return get(ix, iy, iz);
341 }

372 __m128 CubicInterpolate(__m128 p0,__m128 p1,__m128 p2,__m128 p3,double position) const {
373 __m128 c1 = _mm_set1_ps (1/2.);
374 __m128 c2 = _mm_set1_ps (3/2.);
375 __m128 c3 = _mm_set1_ps (2.);
376 __m128 c4 = _mm_set1_ps (5/2.);
377
378 __m128 pos = _mm_set1_ps (position);
379 __m128 pos2 = _mm_set1_ps (position*position);
380 __m128 pos3 = _mm_set1_ps (position*position*position);
381
387 __m128 term = _mm_mul_ps(_mm_sub_ps(_mm_mul_ps(c1,p2),_mm_mul_ps(c1,p0)),pos);
388 __m128 term2 = _mm_mul_ps(_mm_sub_ps(_mm_add_ps(p0,_mm_mul_ps(c3,p2)),_mm_add_ps(_mm_mul_ps(c4,p1),_mm_mul_ps(c1,p3))),pos2);
389 __m128 term3 = _mm_mul_ps(_mm_sub_ps(_mm_add_ps(_mm_mul_ps(c2,p1),_mm_mul_ps(c1,p3)),_mm_add_ps(_mm_mul_ps(c1,p0),_mm_mul_ps(c2,p2))),pos3);
390 __m128 res = _mm_add_ps(_mm_add_ps(_mm_add_ps(term3,term2),term),p1);
391 return res;
392 }

395 Vector3f tricubicInterpolate(Vector3f, const Vector3d &position) const {
396 #ifdef HAVE_SIMD
397 // position on a unit grid
398 Vector3d r = (position - gridOrigin) / spacing;
399
400 int iX0, iY0, iZ0;
401 iX0 = floor(r.x);
402 iY0 = floor(r.y);
403 iZ0 = floor(r.z);
404
405 double fX, fY, fZ;
406 fX = r.x - iX0;
407 fY = r.y - iY0;
408 fZ = r.z - iZ0;
409
410 int nrCubicInterpolations = 4;
411 __m128 interpolateVaryX[nrCubicInterpolations];
412 __m128 interpolateVaryY[nrCubicInterpolations];
413 __m128 interpolateVaryZ[nrCubicInterpolations];
415 for (int iLoopX = -1; iLoopX < nrCubicInterpolations-1; iLoopX++) {
416 for (int iLoopY = -1; iLoopY < nrCubicInterpolations-1; iLoopY++) {
417 for (int iLoopZ = -1; iLoopZ < nrCubicInterpolations-1; iLoopZ++) {
418 if (reflective)
419 interpolateVaryZ[iLoopZ+1] = simdreflectiveGet(iX0+iLoopX, iY0+iLoopY, iZ0+iLoopZ);
420 else
421 interpolateVaryZ[iLoopZ+1] = simdperiodicGet(iX0+iLoopX, iY0+iLoopY, iZ0+iLoopZ);
422 }
423 interpolateVaryY[iLoopY+1] = CubicInterpolate(interpolateVaryZ[0], interpolateVaryZ[1], interpolateVaryZ[2], interpolateVaryZ[3], fZ);
424 }
425 interpolateVaryX[iLoopX+1] = CubicInterpolate(interpolateVaryY[0], interpolateVaryY[1], interpolateVaryY[2], interpolateVaryY[3], fY);
426 }
427 __m128 result = CubicInterpolate(interpolateVaryX[0], interpolateVaryX[1], interpolateVaryX[2], interpolateVaryX[3], fX);
428 return convertSimdToVector3f(result);
429 #else // HAVE_SIMD
430 throw std::runtime_error( "Tried to use tricubic Interpolation without SIMD_EXTENSION. SIMD Optimization is neccesary for tricubic interpolation of vector grids.\n");
431 #endif // HAVE_SIMD
432 }

476 T trilinearInterpolate(const Vector3d &position) const {
478 Vector3d r = (position - gridOrigin) / spacing;
479
482 int iX0, iX1, iY0, iY1, iZ0, iZ1;
483 double resX, resY, resZ, fX0, fY0, fZ0;
484
485 if (reflective) {
486 reflectiveClamp(r.x, Nx, iX0, iX1, resX);
487 reflectiveClamp(r.y, Ny, iY0, iY1, resY);
488 reflectiveClamp(r.z, Nz, iZ0, iZ1, resZ);
489 fX0 = resX - floor(resX);
490 fY0 = resY - floor(resY);
491 fZ0 = resZ - floor(resZ);
492 } else {
493 periodicClamp(r.x, Nx, iX0, iX1);
494 periodicClamp(r.y, Ny, iY0, iY1);
495 periodicClamp(r.z, Nz, iZ0, iZ1);
496 fX0 = r.x - floor(r.x);
497 fY0 = r.y - floor(r.y);
498 fZ0 = r.z - floor(r.z);
499 }
500
502 double fX1 = 1 - fX0;
503 double fY1 = 1 - fY0;
504 double fZ1 = 1 - fZ0;
505
507 T b(0.);
508 b += get(iX0, iY0, iZ0) * fX1 * fY1 * fZ1;
509 b += get(iX1, iY0, iZ0) * fX0 * fY1 * fZ1;
510 b += get(iX0, iY1, iZ0) * fX1 * fY0 * fZ1;
511 b += get(iX0, iY0, iZ1) * fX1 * fY1 * fZ0;
512 b += get(iX1, iY0, iZ1) * fX0 * fY1 * fZ0;
513 b += get(iX0, iY1, iZ1) * fX1 * fY0 * fZ0;
514 b += get(iX1, iY1, iZ0) * fX0 * fY0 * fZ1;
515 b += get(iX1, iY1, iZ1) * fX0 * fY0 * fZ0;
516
517 return b;
518 }
