69// - input: function over which to integrate, integration limits A and B
70// - output: 8-points Gauß-Legendre integral
71static const double X[8] = {.0950125098, .2816035507, .4580167776, .6178762444, .7554044083, .8656312023, .9445750230, .9894009349};
72static const double W[8] = {.1894506104, .1826034150, .1691565193, .1495959888, .1246289712, .0951585116, .0622535239, .0271524594};
73template<typename Integrand>
74double gaussInt(Integrand&& integrand, double A, double B) {
75 const double XM = 0.5 * (B + A);
76 const double XR = 0.5 * (B - A);
77 double SS = 0.;
78 for (int i = 0; i < 8; ++i) {
79 double DX = XR * X[i];
80 SS += W[i] * (integrand(XM + DX) + integrand(XM - DX));
81 }
82 return XR * SS;
83}
